- 直接将索引i寄存在对象中，代码如下：

Html结构：

```html
    <ul id="test">
        <li>0</li>
        <li>1</li>
        <li>2</li>
        <li>3</li>
    </ul>
```

JS：

```javascript
    window.onload = function () {
        var oUl = document.getElementById('test'),
            oLi = oUl.getElementsByTagName('li');
        for (var i = 0,len = oLi.length; i < len; i++) {
            oLi[i].idx = i;
            oLi[i].onclick = function () {
                alert(this.idx)
            }
        }
    }
```

- 利用JS闭包实现，代码如下：

Html结构：

```html
    <ul id="test">
        <li>0</li>
        <li>1</li>
        <li>2</li>
        <li>3</li>
    </ul>
```

JS：
```javascript
    window.onload = function () {
        var oUl = document.getElementById('test'),
            oLi = oUl.getElementsByTagName('li');
        for (var i = 0,len = oLi.length; i < len; i++) {       
            (function (n) {
                oLi[i].onclick = function () {
                    alert(n)
                }
            })(i)
        }
    }
```
